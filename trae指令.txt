1. 创建字体加载锁 Hook src/lib/hooks/useFontReady.ts
该 Hook 用于等待文档字体加载，超时后强制就绪。

typescript
import { useEffect, useState } from 'react';

export function useFontReady(timeoutMs = 3000): boolean {
  const [ready, setReady] = useState(false);

  useEffect(() => {
    let timeout: NodeJS.Timeout;
    // 等待所有字体加载
    document.fonts.ready.then(() => setReady(true));
    timeout = setTimeout(() => setReady(true), timeoutMs); // 超时强制就绪
    return () => clearTimeout(timeout);
  }, [timeoutMs]);

  return ready;
}
2. 创建渲染锁 Hook src/lib/hooks/useRenderLock.ts
该 Hook 用于监控渲染过程，超时后触发降级。

typescript
import { useRef, useCallback } from 'react';

export function useRenderLock(timeoutMs = 5000) {
  const timerRef = useRef<NodeJS.Timeout>();

  const startRender = useCallback((onTimeout: () => void) => {
    if (timerRef.current) clearTimeout(timerRef.current);
    timerRef.current = setTimeout(() => {
      onTimeout(); // 超时回调，用于降级
    }, timeoutMs);
  }, [timeoutMs]);

  const finishRender = useCallback(() => {
    if (timerRef.current) {
      clearTimeout(timerRef.current);
      timerRef.current = undefined;
    }
  }, []);

  return { startRender, finishRender };
}
3. 创建 AbcRenderer 组件 src/components/song/AbcRenderer.tsx
这是核心组件，负责渲染五线谱并集成防御机制。目前不实现指法图注入。

tsx
'use client';

import { useEffect, useRef, useState } from 'react';
import * as abcjs from 'abcjs';
import { useFontReady } from '@/lib/hooks/useFontReady';
import { useRenderLock } from '@/lib/hooks/useRenderLock';
import Skeleton from '@/components/ui/Skeleton';

type AbcRendererProps = {
  abcString: string;          // ABC 源码
  instrumentId: string;       // 乐器 ID（暂未使用，但保留）
  onRenderStart?: () => void;
  onRenderComplete?: () => void;
};

export default function AbcRenderer({
  abcString,
  instrumentId,
  onRenderStart,
  onRenderComplete,
}: AbcRendererProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const [hasError, setHasError] = useState(false);
  const [isTimeout, setIsTimeout] = useState(false);
  const fontReady = useFontReady(3000);
  const { startRender, finishRender } = useRenderLock(5000);

  // 降级处理函数：超时或错误时调用
  const handleFallback = () => {
    setIsTimeout(true);
    setHasError(false); // 超时不算错误，只是不渲染指法图
    // 可以在这里清理可能的部分渲染
    if (containerRef.current) {
      containerRef.current.innerHTML = ''; // 清空以防残留
    }
    // 通知父组件完成（降级完成也算完成）
    onRenderComplete?.();
  };

  useEffect(() => {
    if (!fontReady) return; // 字体未就绪，继续等待

    // 重置状态
    setHasError(false);
    setIsTimeout(false);
    onRenderStart?.();

    // 启动渲染锁
    startRender(handleFallback);

    // 执行 abcjs 渲染
    if (containerRef.current && abcString) {
      try {
        // 清空容器
        containerRef.current.innerHTML = '';

        // 调用 abcjs 渲染
        abcjs.renderAbc(containerRef.current, abcString, {
          responsive: 'resize',
          add_classes: true,      // 添加 CSS 类名，便于后续定位
          staffwidth: 600,        // 五线谱宽度（可调）
          paddingtop: 20,
          paddingbottom: 30,
        });

        // 渲染完成，结束锁
        finishRender();
        onRenderComplete?.();
      } catch (err) {
        console.error('abcjs 渲染错误:', err);
        setHasError(true);
        finishRender();
        onRenderComplete?.();
      }
    }

    // 清理函数
    return () => {
      finishRender();
      if (containerRef.current) {
        containerRef.current.innerHTML = '';
      }
    };
  }, [abcString, fontReady, startRender, finishRender, onRenderStart, onRenderComplete]);

  // 如果字体未就绪，显示骨架屏
  if (!fontReady) {
    return <Skeleton />;
  }

  // 如果发生错误，显示错误信息
  if (hasError) {
    return (
      <div className="p-8 text-center text-wood-dark bg-red-50 rounded-lg">
        ❌ 乐谱加载失败，请稍后重试。
      </div>
    );
  }

  // 如果超时（但未错误），显示纯五线谱（不注入指法图），但保留容器
  return (
    <div className="relative">
      {isTimeout && (
        <div className="absolute top-2 right-2 bg-yellow-100 text-yellow-800 text-xs px-2 py-1 rounded">
          加载超时，已显示基础谱面
        </div>
      )}
      <div ref={containerRef} className="abc-paper"></div>
    </div>
  );
}
